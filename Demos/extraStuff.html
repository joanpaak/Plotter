More examples

Rectangles, polygons, text, changing colours, rotation, interaction etc.

Drawing rectangles

Three rectangles are drawn. Note that you can pass an object containing SVG attributes as the third argument, these attributes will be applied to the element when it is drawn.

            let plot = new Plot(
                document.getElementById("test_1"),
                null,
                {
                    xlim : [-10, 10],
                    ylim : [-10, 10]
                }
            );

            plot.drawRect(0, 0, 5, 5);
            plot.drawRect(5, 5, 1, 1, {fill : "red"});
            plot.drawRect(2, 2, 2, 2, {fill : "rgb(40 140 200)", rotation : "45"});

Making the plots interactive

It is relatively easy to make the plots interactive via Javascript because when you draw something to the plot, the drawn object is returned. You can catch the object, assign it to a variable and the do with it whatever you just want...

Note that if the element's coordinates are changed while it is rotated, and it is then rotated again, this will make the element jump around a bit. This is because when the element is moved, the center point of the rotation should move with it. Obviously that is something that is not currently implemented.

To get the coordinates right, you can use the plot-objects transform methods as in the example.

            let plot = new Plot(
                document.getElementById("test_3"),
                null,{
                    xlim : [-10, 10],
                    ylim : [-10, 10]
                }
            );

            let text = plot.addText("Don't spin me, I'll puke!", 0, 0);
            
            let rotSlider = document.getElementById("test_3_rotation");
            let xSlider = document.getElementById("test_3_x");
            let ySlider = document.getElementById("test_3_y");
            
            rotSlider.addEventListener("input", () => {
                plot.rotateElement(text, parseInt(rotSlider.value));
            });


            xSlider.addEventListener("input", () => {
                text.setAttribute("x", plot.transformXCoordinate(xSlider.value));
            });

            ySlider.addEventListener("input", () => {
                text.setAttribute("y", plot.transformYCoordinate(ySlider.value));
            });

Polygons

Polygons, of course! Everyone loves polygons. Note that in the code below the polygons are saved into objects: when you call a drawing method, it returns the SVG object so that it can be easily manipulated. Of course this can be omitted and you can call the methods without explicitly saving the returned object.

            let plot = new Plot(
                document.getElementById("test_4"),
                null,
                {
                    xlim : [-2, 2],
                    ylim : [-2, 2]
                }
            );

            let xcoords = [-1, 0, 1, 0];
            let ycoords = [0, 1, 0, -1];

            let poly = plot.drawPolygon(
                xcoords, 
                ycoords, 
                {
                    fill : "rgb(255, 230, 25)",
                    opacity : "0.5", 
                    stroke : "black"
                });
            
            let poly2 = plot.drawPolygon(
                [-0.5, 0.5, 1.5, 0.5],
                [1, 2, 1, 0],
                {
                    opacity : "0.5"
                }
            )

Multicoloured balls

            let plot = new Plot(
                document.getElementById("test_5"),
                null,
                {
                    xlim : [0, 255],
                    ylim : [0, 100],
                    xlab : "Hue",
                    ylab : "Saturation (%)"
                }
            );

            let n = 500;
            
            for(let i = 0; i < n; i++){
                let hue = Math.random() * 255;
                let saturation = Math.random() * 100;

                plot.drawPoint(
                    hue,
                    saturation,
                    {
                        fill : `hsl(${hue}, ${saturation}%, 50%)`
                    });
            }

Plot a function using .drawCurve(f, min, max, opt)

            function xSquared(x){
                return x * x - 2;
            }

            let plot = new Plot(
                document.getElementById("test_6"),
                null,
                {
                    xlim : [-10, 10],
                    ylim : [-5, 5]

                }
            )

            plot.hline(0, {"stroke-dasharray" : [2, 2]});
            plot.vline(0, {"stroke-dasharray" : [2, 2]});
            plot.drawCurve(xSquared, -5, 5);
